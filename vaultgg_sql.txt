CREATE TABLE dbProj_categories (
    cat_id     INT AUTO_INCREMENT PRIMARY KEY,
    name       VARCHAR(50)  NOT NULL,
    slug       VARCHAR(50)  NOT NULL UNIQUE,
    color      VARCHAR(20)  NOT NULL DEFAULT '#00f0ff',
    emoji      VARCHAR(10)  NOT NULL DEFAULT '🎮',
    sort_order INT          NOT NULL DEFAULT 0
);

CREATE TABLE dbProj_users (
    user_id    INT AUTO_INCREMENT PRIMARY KEY,
    username   VARCHAR(50)  NOT NULL UNIQUE,
    email      VARCHAR(100) NOT NULL UNIQUE,
    password   VARCHAR(255) NOT NULL,
    first_name VARCHAR(50)  NOT NULL,
    last_name  VARCHAR(50)  NOT NULL,
    role       ENUM('visitor','creator','admin') NOT NULL DEFAULT 'visitor',
    is_active  TINYINT(1)   NOT NULL DEFAULT 1,
    created_at DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    last_login DATETIME     DEFAULT NULL
);

CREATE TABLE dbProj_accounts (
    account_id   INT AUTO_INCREMENT PRIMARY KEY,
    creator_id   INT            NOT NULL,
    cat_id       INT            NOT NULL,
    title        VARCHAR(150)   NOT NULL,
    slug         VARCHAR(160)   NOT NULL UNIQUE,
    short_desc   VARCHAR(300)   NOT NULL,
    description  TEXT           NOT NULL,
    price        DECIMAL(10,2)  NOT NULL,
    original_val DECIMAL(10,2)  NOT NULL,
    rank_label   VARCHAR(50)    NOT NULL DEFAULT 'Unranked',
    level_val    INT            DEFAULT NULL,
    skins_count  INT            DEFAULT NULL,
    wins_count   INT            DEFAULT NULL,
    perks        TEXT           DEFAULT NULL,
    tags         VARCHAR(300)   DEFAULT NULL,
    is_hot       TINYINT(1)     NOT NULL DEFAULT 0,
    status       ENUM('draft','published','removed','sold') NOT NULL DEFAULT 'draft',
    view_count   INT            NOT NULL DEFAULT 0,
    created_at   DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at   DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    published_at DATETIME       DEFAULT NULL,
    FOREIGN KEY (creator_id) REFERENCES dbProj_users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (cat_id)     REFERENCES dbProj_categories(cat_id) ON DELETE RESTRICT
);

ALTER TABLE dbProj_accounts

CREATE TABLE dbProj_comments (
    comment_id INT AUTO_INCREMENT PRIMARY KEY,
    account_id INT  NOT NULL,
    user_id    INT  NOT NULL,
    body       TEXT NOT NULL,
    is_removed TINYINT(1) NOT NULL DEFAULT 0,
    created_at DATETIME   NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (account_id) REFERENCES dbProj_accounts(account_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id)    REFERENCES dbProj_users(user_id) ON DELETE CASCADE
);

CREATE TABLE dbProj_ratings (
    rating_id  INT AUTO_INCREMENT PRIMARY KEY,
    account_id INT     NOT NULL,
    user_id    INT     NOT NULL,
    score      TINYINT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uq_user_account (account_id, user_id),
    FOREIGN KEY (account_id) REFERENCES dbProj_accounts(account_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id)    REFERENCES dbProj_users(user_id) ON DELETE CASCADE
);

CREATE TABLE dbProj_activity_log (
    log_id      INT AUTO_INCREMENT PRIMARY KEY,
    action      VARCHAR(50)  NOT NULL,
    entity_type VARCHAR(30)  NOT NULL,
    entity_id   INT          NOT NULL,
    user_id     INT          DEFAULT NULL,
    note        VARCHAR(300) DEFAULT NULL,
    logged_at   DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE dbProj_wishlist (
    wishlist_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id     INT      NOT NULL,
    account_id  INT      NOT NULL,
    created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uq_user_account (user_id, account_id),
    FOREIGN KEY (user_id)    REFERENCES dbProj_users(user_id)    ON DELETE CASCADE,
    FOREIGN KEY (account_id) REFERENCES dbProj_accounts(account_id) ON DELETE CASCADE
);

CREATE TABLE dbProj_purchases (
    purchase_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id     INT            NOT NULL,
    account_id  INT            NOT NULL,
    amount_paid DECIMAL(10,2)  NOT NULL,
    status      ENUM('active','refunded','cancelled') NOT NULL DEFAULT 'active',
    created_at  DATETIME       NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id)    REFERENCES dbProj_users(user_id)    ON DELETE CASCADE,
    FOREIGN KEY (account_id) REFERENCES dbProj_accounts(account_id) ON DELETE CASCADE
);

-- PROCEDURE 1: Run this alone, click Go
CREATE PROCEDURE sp_report_popular(
    IN p_date_from DATE,
    IN p_date_to   DATE,
    IN p_limit     INT
)
BEGIN
    SELECT a.account_id, a.title, a.price, a.view_count,
           c.name AS category,
           u.username AS creator,
           COALESCE(AVG(r.score), 0) AS avg_rating,
           COUNT(DISTINCT cm.comment_id) AS comment_count,
           a.published_at
    FROM dbProj_accounts a
    JOIN dbProj_categories c ON a.cat_id = c.cat_id
    JOIN dbProj_users u ON a.creator_id = u.user_id
    LEFT JOIN dbProj_ratings r ON a.account_id = r.account_id
    LEFT JOIN dbProj_comments cm ON a.account_id = cm.account_id
                                 AND cm.is_removed = 0
    WHERE a.status = 'published'
      AND (p_date_from IS NULL OR DATE(a.published_at) >= p_date_from)
      AND (p_date_to   IS NULL OR DATE(a.published_at) <= p_date_to)
    GROUP BY a.account_id
    ORDER BY a.view_count DESC, avg_rating DESC
    LIMIT p_limit;
END


-- PROCEDURE 2: Run this alone, click Go
CREATE PROCEDURE sp_report_by_creator(IN p_user_id INT)
BEGIN
    SELECT a.account_id, a.title, a.status, a.price,
           a.view_count, a.created_at, a.published_at,
           c.name AS category,
           COALESCE(AVG(r.score), 0) AS avg_rating,
           COUNT(DISTINCT cm.comment_id) AS comment_count
    FROM dbProj_accounts a
    JOIN dbProj_categories c ON a.cat_id = c.cat_id
    LEFT JOIN dbProj_ratings r ON a.account_id = r.account_id
    LEFT JOIN dbProj_comments cm ON a.account_id = cm.account_id
                                 AND cm.is_removed = 0
    WHERE a.creator_id = p_user_id
    GROUP BY a.account_id
    ORDER BY a.created_at DESC;
END

SCRIPT 5 – TRIGGERS
Use phpMyAdmin Triggers Tab method (not SQL tab)
Go to table → Triggers tab → Add Trigger

-- TRIGGER 1
-- Table: dbProj_accounts
-- Name: trg_account_published
-- Time: AFTER  Event: UPDATE
-- Definition (paste in Definition box):

IF OLD.status != 'published' AND NEW.status = 'published' THEN
    INSERT INTO dbProj_activity_log
        (action, entity_type, entity_id, user_id, note)
    VALUES
        ('PUBLISHED', 'account', NEW.account_id, NEW.creator_id,
         CONCAT('Account published: ', NEW.title));
END IF;


-- TRIGGER 2
-- Table: dbProj_accounts
-- Name: trg_account_removed
-- Time: AFTER  Event: UPDATE
-- Definition (paste in Definition box):

IF OLD.status != 'removed' AND NEW.status = 'removed' THEN
    INSERT INTO dbProj_activity_log
        (action, entity_type, entity_id, note)
    VALUES
        ('REMOVED', 'account', NEW.account_id,
         CONCAT('Account removed: ', NEW.title));
END IF;


-- TRIGGER 3
-- Table: dbProj_users
-- Name: trg_user_registered
-- Time: AFTER  Event: INSERT
-- Definition (paste in Definition box):

INSERT INTO dbProj_activity_log
    (action, entity_type, entity_id, note)
VALUES
    ('REGISTERED', 'user', NEW.user_id,
     CONCAT('New user registered: ', NEW.username));
